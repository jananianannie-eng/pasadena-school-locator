// ---------------- GEOCODE ----------------

async function geocode(address) {

  const queries = [

    address + ", Pasadena, CA",

    address + ", CA",

    address

  ];


  for (let q of queries) {

    const res = await fetch(

      "https://nominatim.openstreetmap.org/search?format=json&limit=1&q=" +

      encodeURIComponent(q)

    );


    const data = await res.json();

    if (data && data.length > 0) return data[0];

  }


  return null;

}


// ---------------- DISTANCE ----------------

function getDistance(a, b) {

  return map.distance(a, b);

}


// ---------------- FIND NEAREST ----------------

function findNearest(lat, lng, level) {

  let nearest = null;

  let minDist = Infinity;


  for (let s of schools) {

    if (s.level !== level) continue;


    const dist = getDistance([lat, lng], [s.lat, s.lng]);


    if (dist < minDist) {

      minDist = dist;

      nearest = s;

    }

  }


  return { nearest, minDist };

}


// ---------------- MAIN ----------------

async function findSchool() {

  const address = document.getElementById("address").value;

  const level = document.getElementById("level").value;


  if (!address) {

    alert("Please enter an address");

    return;

  }


  document.getElementById("box").innerHTML = "Searching...";


  const geo = await geocode(address);


  if (!geo) {

    document.getElementById("box").innerHTML =

      "❌ Address could not be located";

    return;

  }


  const lat = parseFloat(geo.lat);

  const lng = parseFloat(geo.lon);


  if (isNaN(lat) || isNaN(lng)) {

    document.getElementById("box").innerHTML =

      "❌ Invalid location data";

    return;

  }


  // Map marker for user

  if (userMarker) map.removeLayer(userMarker);


  userMarker = L.marker([lat, lng])

    .addTo(map)

    .bindPopup("Search Location")

    .openPopup();


  map.setView([lat, lng], 13);


  // Find school

  const result = findNearest(lat, lng, level);


  if (!result.nearest) {

    document.getElementById("box").innerHTML =

      "No school found.";

    return;

  }


  if (schoolMarker) map.removeLayer(schoolMarker);


  schoolMarker = L.marker([result.nearest.lat, result.nearest.lng])

    .addTo(map)

    .bindPopup(result.nearest.name)

    .openPopup();


  const miles = (result.minDist * 0.000621371).toFixed(2);


  document.getElementById("box").innerHTML =

    "<b>Nearest School:</b> " + result.nearest.name + "<br>" +

    "<b>Level:</b> " + result.nearest.level + "<br>" +

    "<b>Distance:</b> " + miles + " miles";

}
